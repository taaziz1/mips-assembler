# Translate the follow C Code to MIPS
# C[0] = A[0]+B[0];
# C[1] = A[1]+B[1];
# C[2] = A[2]+B[2];

.data
  A:  .byte 1,2,3   # char A[] = { ... };
  B:  .byte 3,2,1   # char B[] = { ... };
  C:  .byte 0,0,0   # char C[] = { ... };

.text
	LA $1,A
	LA $2,B
	LA $3,C

	LB $4 , 0($1)
	LB $5 , 0($2)
	ADD $6, $4 , $5
	SB  $6, 0($3)

	LB     $4 , 1($1)
	LB     $5 , 1($2)
	ADD    $6, $4 , $5
	SB     $6, 1($3)

	LB     $4 , 2($1)
	LB     $5 , 2($2)
	ADD    $6, $4 , $5
	SB     $6, 2($3)
	stop
